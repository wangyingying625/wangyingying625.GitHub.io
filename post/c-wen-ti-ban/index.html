    如果length小或者相等,那么把数组的第k个位置赋值'\0'

#include <stdio.h>
#include <string.h>
int main(){
    char * p[]={"hi","helo","java","C++"};
    int i;
    for(i=0;i<4;i++){
        printf("%c:",*p[i]);
        printf("%s\n",*(p+i));
    }
    return 0;
}


指针数组程序解读:
首先 char * p[]是一个指针数组,相当于一个二维数组(数组的每一个内容都是指针)
p[0]是指向字符串hi的指针,p[1]是指向字符串helo的指针...
但是p[0]的内容是所指向字符串的首字母
(p+1)的内容才是字符串

#include <stdio.h>
#include <stdlib.h>
typedef struct node{
    int num;
    double height;
    struct node * next;
}NODE;
NODE * CreateLink(){
    NODE *head=NULL ;
    NODE *p,*q;
    while (1) {
        NODE * node =(NODE *)malloc(sizeof(NODE));
        scanf("%d",&node->num);
        if(node->num==-1)
            break;
        printf("请输入shengao\n");
        scanf("%lf",&node->height);
        node->next=NULL;
        
        if(head==NULL){
            head=node;
        }
        else{
            q=head;
            if(q&&q->height>node->height){
                node->next=q;
                head=node;
            }
            else{
                while(q&&q->height<node->height){
                    p=q;
                    q=q->next;
                }
                node->next=q;
                p->next=node;
            }
        }
    }
    return head;
};
NODE *MergeLink(NODE *w,NODE *m){
    NODE *head=NULL;
    NODE * node =(NODE *)malloc(sizeof(NODE));
    NODE *p,*q;
    p=w;
    q=m;
    while(p!=NULL || q!=NULL){
        if(p!=NULL&& q!=NULL && (p->height)<=(q->height)){
            if(head==NULL){
                node=p;
                head=node;
                p=p->next;
            }
            else{
                node->next=p;
                node=node->next;
                p=p->next;
            }
        }
        else if(p!=NULL&& q!=NULL && p->height>q->height){
            if(head==NULL){
                node=q;
                head=node;
                q=q->next;
            }
            else{
                node->next=q;
                node=node->next;
                q=q->next;
            }
        }
        else if(p==NULL&& q!=NULL){
            node->next=q;
            break;
        }
        else if(p!=NULL&& q==NULL){
            node->next=p;
            break;
        }
    }
    return head;
};
void output(NODE *h){
    while(h!=NULL){
        printf("%lf\n",h->height);
        h=h->next;
    }
}
int main(){
    NODE *H,*H_M,*H_F;
    printf("请输入男生信息\n");
    H_M = CreateLink();
    printf("请输入女生信息\n");
    H_F=CreateLink();
    H=MergeLink(H_M,H_F);
    output(H);
    return 0;
}
